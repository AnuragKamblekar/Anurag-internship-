def preprocess_sensor(df, rename_map, ref_col='seconds_elapsed'):
    """Rename axis columns, drop duplicates, sort by time."""
    df = df.rename(columns=rename_map)
    df = df.sort_values(ref_col).drop_duplicates(subset=[ref_col]).reset_index(drop=True)
    return df

accel_df = preprocess_sensor(accel_df, {'x': 'acc_x', 'y': 'acc_y', 'z': 'acc_z'})

def merge_nearest(base, other, on='seconds_elapsed', tolerance=1.0):
    merged = pd.merge_asof(base.sort_values(on), other.sort_values(on), on=on, direction='nearest', tolerance=tolerance)
    return merged

merged = merge_nearest(ref, gyro_df[['seconds_elapsed','gyr_x','gyr_y','gyr_z']])

def make_sequences(X, y, seq_len):
    Xs, ys = [], []

        ys.append(y[i+seq_len])
    return np.array(Xs), np.array(ys)
